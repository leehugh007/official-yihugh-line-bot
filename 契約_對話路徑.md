### 3.3 `validateAiOutput` 防禦層（v4 強化）

**v4 新增 top-level 防禦**（第三輪 Feasibility 指出）：

```js
function validateAiOutput(stage, path, output) {
  if (!output || typeof output !== 'object' || Array.isArray(output)) {
    return { ok: false, reason: 'output_not_object' };
  }
  if (!output.confidence) return { ok: false, reason: 'missing_confidence' };

  // v4 新：current/target_weight 型別檢查
  if (output.current_weight != null && typeof output.current_weight !== 'number') {
    return { ok: false, reason: 'current_weight_not_number' };
  }
  if (output.target_weight != null && typeof output.target_weight !== 'number') {
    return { ok: false, reason: 'target_weight_not_number' };
  }

  // path enum
  if (output.path != null && !['healthCheck','rebound','postpartum','eatOut','other'].includes(output.path)) {
    return { ok: false, reason: `invalid_path_${output.path}` };
  }

  // conditions coerce
  if (output.conditions != null) {
    if (typeof output.conditions === 'string') output.conditions = [output.conditions];
    if (!Array.isArray(output.conditions)) return { ok: false, reason: 'conditions_not_array' };
    const validConditions = getValidConditions(stage, output.path);
    const invalid = output.conditions.filter(c => !validConditions.includes(c));
    if (invalid.length > 0) return { ok: false, reason: `invalid_conditions_${invalid.join(',')}` };
  }

  // v4 新：ai_tags top-level 防禦
  if (output.ai_tags != null) {
    if (typeof output.ai_tags !== 'object' || Array.isArray(output.ai_tags)) {
      output.ai_tags = {}; // 整個丟棄，不 crash
    } else {
      const a = output.ai_tags;
      for (const k of ['pain_points','hesitations','attentions']) {
        if (a[k] == null) continue;
        if (typeof a[k] === 'string') a[k] = [a[k]];
        if (!Array.isArray(a[k])) { a[k] = []; continue; }
        // v4 新：處理 object 陣列（AI 回 [{value:"..."}]) + 純 string 陣列混合
        a[k] = a[k]
          .map(x => typeof x === 'string' ? x : (x?.value || null))
          .filter(x => typeof x === 'string');
      }
      if (a.intent != null && !['high','medium','low'].includes(a.intent)) {
        a.intent = 'medium'; // 降級不 reject
      }
    }
  }

  return { ok: true, output, fallback: output.confidence === 'low' };
}
```

import { isDynamic } from './dynamic-templates.js';  // v6: ESM

async function renderTemplate(template, user, aiOutput = null) {
  if (isDynamic(template.id)) {
    if (!aiOutput?.feedback_text) throw new Error(`DYNAMIC ${template.id} missing aiOutput`);
    return aiOutput.feedback_text;
  }
  let text = template.message_template;
  const vars = {
    '{diff}': () => Math.abs(user.target_weight - user.current_weight),
    '{current}': () => user.current_weight,
    '{target}': () => user.target_weight,
  };
  for (const [p, get] of Object.entries(vars)) {
    if (text.includes(p)) {
      const val = get();
      if (val == null) throw new Error(`${template.id} needs ${p} null`);
      text = text.replaceAll(p, String(val));
    }
  }
  return text;
}

### 5.10.1 Cron 時區判斷（v5 修正重複觸發）

現有 `vercel.json` cron 是 `*/10 * * * *`（每 10 分鐘跑），全部用 **UTC**。cron B/C 要「每日台北 03:00 = UTC 19:00 跑一次」，但每 10 分鐘 cron 會在 UTC 19:00-19:50 連跑 6 次。

**解法**：在 processor 開頭判「是否剛好在 UTC 19:00-19:09 區間」，其他時段 skip：

```js
async function processAiTagsFreshness() {
  const now = new Date();
  // 台北 03:00 = UTC 19:00。只在 UTC 19:00-19:09 跑（每日第一次觸發），避免 6 次連跑
  if (now.getUTCHours() !== 19 || now.getUTCMinutes() >= 10) {
    return { skipped: 'not_daily_window' };
  }
  // ... 實際 cron B 邏輯
}
// processStage5Rescue 同樣邏輯（或放在不同時段，例如 UTC 20:00 = 台北 04:00 分離）
```

**為什麼用 `getUTCHours` 而非 `getHours`**：Vercel serverless 跑 UTC，但本地測試 `getHours()` 會用機器 timezone → 行為不一致。明寫 `getUTCHours` 避免歧義。

async function triggerHandoff(userId, reason, replyToken) {
  // 1. 讀 current ai_tags
  const { data: user } = await supabase
    .from('official_line_users')
    .select('ai_tags, handoff_triggered_at')
    .eq('line_user_id', userId).single();

  // 2. 避免重複（v6 修正：同 reason 1h 內不觸發；不同 reason 補 notify 但不動 handoff_*）
  if (user?.handoff_triggered_at && user?.handoff_reason === reason) {
    const elapsed = Date.now() - new Date(user.handoff_triggered_at).getTime();
    if (elapsed < 3600_000) return { skipped: 'same_reason_within_1h' };
  }
  const isRepeatDifferentReason = user?.handoff_triggered_at && user?.handoff_reason !== reason;

  // v6 分支：isRepeatDifferentReason 只推 notify，不動任何欄位
  if (isRepeatDifferentReason) {

    // 保留第一次 Handoff 記錄，只告訴婉馨有新訊號
    return { ok: true, mode: 'supplement_notify' };
  }

  // 正常第一次 Handoff 路徑（或 reason 相同但 >1 小時）

  // 3. 合併 ai_tags 清 intent='high' → medium
  const newAiTags = mergeByOp(user?.ai_tags || {}, {
    意願: 'medium',
    _op: 'overwrite'
  });

  // 4. 單次 UPDATE 同步寫 stage + handoff + ai_tags
  const now = new Date().toISOString();
  await supabase.from('official_line_users').update({
    path_stage: 5,
    path_stage_updated_at: now,
    handoff_triggered_at: now,
    handoff_reason: reason,
    handoff_rescue_notified: false,
    ai_tags: newAiTags,
    ai_tags_updated_at: now
  }).eq('line_user_id', userId);

  // 5. 推 notify
  await pushHandoffNotify(userId, reason);

  // 6. log chat_history（Phase 3 建表後）

## 9. 寫入規則（Data Lifecycle）

### 9.1 `updateAiTags(userId, patch)` + `mergeByOp` 完整實作（v4 明文）

```js
// lib/users.js 新增

const EN_TO_ZH = {
  pain_points: '痛點',
  hesitations: '猶豫',
  intent: '意願',
  attentions: '關注',
};

const ARRAY_FIELDS = ['痛點', '猶豫', '關注'];
// v5: 白名單 gate，擋 AI 幻覺傳 `weight: 80` 或 `mood: 'happy'` 污染 ai_tags
const ALLOWED_KEYS = new Set([
  ...ARRAY_FIELDS,
  '意願',
  'guide_miss_count',
  'retry_count_q1', 'retry_count_q3', 'retry_count_q4',
  '_stale',
]);

/**
 * 合併 current ai_tags 和 patch
 * 規則：
 *   - ARRAY_FIELDS（痛點/猶豫/關注）：_op='append' 去重（by value）；_op='overwrite' 替換
 *   - 其他白名單 key（意願/counter/_stale）：永遠覆蓋（忽略 _op）
 *   - 不在白名單的 key：靜默忽略（v5 防 AI 幻覺）
 */
function mergeByOp(current = {}, patch) {
  const { _op = 'append', _from_ai, ...data } = patch;
  const next = { ...current };
  for (const [k, v] of Object.entries(data)) {
    if (!ALLOWED_KEYS.has(k)) continue;  // v5: 白名單外的 key 拒絕寫入
    if (ARRAY_FIELDS.includes(k)) {
      // 陣列：coerce string[] → object[]
      const items = Array.isArray(v) ? v.map(item => {
        if (typeof item === 'string') return { value: item, recorded_at: new Date().toISOString() };
        if (item?.value) return { value: item.value, recorded_at: item.recorded_at || new Date().toISOString() };
        return null;
      }).filter(Boolean) : [];

      if (_op === 'overwrite') {
        next[k] = items;
      } else {
        // append 去重 by value
        const existing = Array.isArray(next[k]) ? next[k] : [];
        const existingValues = new Set(existing.map(x => x.value));
        const newItems = items.filter(x => !existingValues.has(x.value));
        next[k] = [...existing, ...newItems];
      }
    } else {
      // 純量：永遠覆蓋
      next[k] = v;
    }
  }
  return next;
}

async function updateAiTags(userId, patch) {
  // 1. 剝離 meta + 英文→中文 map
  const { _from_ai, _op = 'append', ...rawData } = patch;
  let mapped = rawData;
  if (_from_ai) {
    mapped = Object.fromEntries(
      Object.entries(rawData).map(([k, v]) => [EN_TO_ZH[k] || k, v])
    );
  }

  // 2. read-modify-write
  const { data: user, error: readErr } = await supabase
    .from('official_line_users')
    .select('ai_tags')
    .eq('line_user_id', userId)
    .single();
  if (readErr) return { ok: false, error: readErr.message };

  const merged = mergeByOp(user?.ai_tags || {}, { ...mapped, _op });

  const { error: writeErr } = await supabase
    .from('official_line_users')
    .update({ ai_tags: merged, ai_tags_updated_at: new Date().toISOString() })
    .eq('line_user_id', userId);

  if (writeErr) return { ok: false, error: writeErr.message };
  return { ok: true, data: merged };
}

export { updateAiTags, mergeByOp };  // v5: ESM export，對齊專案 Next.js 14 ESM
```

**Race condition**：放棄 lock（Vercel serverless in-memory lock 無效）。接受「秒內併發漏寫」，Phase 4 AI 下一則訊息重抽補回。若量測 >5% race → 升級 Supabase advisory lock（`SELECT pg_advisory_xact_lock(hashtext(user_id))`），但 Phase 2b **不預先實作**。

**⚠️ v5 誠實標記**：`updateAiTags` / `updatePathStage` 都是 **半 atomic**（read-modify-write 兩個 roundtrip），不是真 atomic。同用戶秒內併發仍可能漏寫。真 atomic 需走 advisory lock 或 optimistic concurrency（`UPDATE ... WHERE ai_tags_updated_at = :read_ts`），**Phase 2b 不做**，下一則訊息 self-heal 可接受。

**Phase 3 webhook caller 必做**：
- 一律 `await` helper 回傳 + 檢查 `.ok`
- `.ok === false` → `console.error(error)` + 繼續流程（不擋用戶）
- 禁止 fire-and-forget（`updateAiTags(...)` 後不 await）— 會吞錯誤

### 9.2 `updatePathStage(userId, newStage, meta)` — v4 atomic 修正

**v3 錯誤**：`updatePathStage` 先 UPDATE stage，再呼叫 `updateAiTags` reset retry_count → 兩次 UPDATE，race window 存在。

**v4 修正**：合併成**單次 UPDATE**，同時寫 path_stage + ai_tags。

```js
async function updatePathStage(userId, newStage, meta = {}) {
  const { path, handoff_reason } = meta;
  const now = new Date().toISOString();

  // 1. 讀 current ai_tags
  const { data: user, error: readErr } = await supabase
    .from('official_line_users')
    .select('ai_tags, handoff_triggered_at')
    .eq('line_user_id', userId)
    .single();
  if (readErr) return { ok: false, error: readErr.message };

  // 2. 決定要 reset 哪個 retry_count
  const resetMap = {
    1: { retry_count_q1: 0 },
    2: { retry_count_q1: 0 },
    3: { retry_count_q3: 0 },
    4: { retry_count_q4: 0 },
  };
  const aiTagsPatch = resetMap[newStage] || {};
  const newAiTags = Object.keys(aiTagsPatch).length > 0
    ? mergeByOp(user?.ai_tags || {}, { ...aiTagsPatch, _op: 'overwrite' })
    : (user?.ai_tags || {});

  // 3. 組 UPDATE
  const update = {
    path_stage: newStage,
    path_stage_updated_at: now,
    ai_tags: newAiTags,
    ai_tags_updated_at: now,
  };
  if (path !== undefined) update.path = path;
  if (newStage === 5) {
    update.handoff_triggered_at = now;
    update.handoff_reason = handoff_reason;
    update.handoff_rescue_notified = false;
  }

  // 4. 單次 UPDATE
  const { error: writeErr } = await supabase
    .from('official_line_users')
    .update(update)
    .eq('line_user_id', userId);

  if (writeErr) return { ok: false, error: writeErr.message };
  return { ok: true, data: update };
}
```

**設計**：
- 單次 SQL UPDATE，atomic
- retry_count reset 從 updateAiTags 呼叫變成 mergeByOp 內部算好一次寫
- Handoff 觸發（stage=5）合併在同一次 UPDATE，包含 `handoff_rescue_notified=false`

### 10.1.1 /api/health endpoint（v5 新增，Step 5 部署驗證用）

**為什麼需要**：若 migration 漏跑，`test_mode` DB 是 null → fallback defaults.js = `true` → 全用戶靜默，沒人發現。/api/health 提供 contract 檢查機制。

**建檔**：`app/api/health/route.js`

```js
import { supabase } from '@/lib/supabase';
import { SETTING_SCHEMA } from '@/lib/official-settings';

export async function GET(request) {
  const url = new URL(request.url);
  const isContractCheck = url.searchParams.get('contract') === 'check';

  if (!isContractCheck) {
    return Response.json({ ok: true, service: 'official-yihugh-line-bot' });
  }

  const expectedVersion = 'v6';  // v6: 跟契約版號同步

  // v6 修正：直查 DB，不透過 getSettingTyped（它會 fallback 到 defaults，永遠不會 null）
  const expectedKeys = Object.keys(SETTING_SCHEMA);
  const { data: settingsRows } = await supabase
    .from('official_settings')
    .select('key, value')
    .in('key', expectedKeys);

  const dbKeys = new Set((settingsRows || []).map(r => r.key));
  const missingSettings = expectedKeys.filter(k => !dbKeys.has(k));
  const dbVersionRow = (settingsRows || []).find(r => r.key === 'contract_version');
  const dbVersion = dbVersionRow?.value || null;

  const testModeRow = (settingsRows || []).find(r => r.key === 'test_mode');
  const testMode = testModeRow?.value === 'true';

  // 查 official_webhook_events 表是否存在（v6 新增）
  const { error: webhookTblErr } = await supabase
    .from('official_webhook_events')
    .select('event_id')
    .limit(1);
  const webhookTableReady = !webhookTblErr;

  return Response.json({
    contract_version_code: expectedVersion,
    contract_version_db: dbVersion,
    mismatch: dbVersion !== expectedVersion,
    missing_settings: missingSettings,
    test_mode: testMode,
    webhook_idempotency_table: webhookTableReady ? 'ok' : 'missing',
    ok: dbVersion === expectedVersion
      && missingSettings.length === 0
      && webhookTableReady,
  });
}
```

### 11.4 升版 Playbook（v4 修正部署順序）

**核心原則**：**migration 必須 add-only**（加新欄位、不改型別、不刪欄位）→ **code 讀寫寬容向下相容**（讀新欄位若 null 有 fallback）。這樣 code 與 DB 的部署順序不重要。

**實作 checklist**：

1. **契約定版**（yi-challenge 三輪通過）
2. **migration 設計審查**：
   - 所有 ALTER TABLE 都是 `ADD COLUMN IF NOT EXISTS`（不改舊欄位）
   - 新欄位有 `DEFAULT` 或 `NULLABLE`
   - 既有 row 不需資料回填（用 DEFAULT 自動填）
3. **部署實務順序（v4 修正）**：
   - Step 1: 本地 `mcp__supabase__apply_migration` 跑 migration → DB schema 升到 vN+1
   - Step 2: 本地驗證 `SELECT` 新欄位存在 + settings 新 key 存在
   - Step 3: `git commit` 新 code + 新契約文件
   - Step 4: `git push main` → Vercel 自動部署 code
   - Step 5: Vercel 部署完成後 curl `/api/health` 確認 Bot 回應正常
   - **重點**：Step 1 先於 Step 4。若 Step 4 先於 Step 1（意外 push），由於 code 讀新欄位若 null → fallback 到 code 常數（defaults.js），Bot **不 crash**，只是功能降級（讀預設值）
4. **Fail-fast 行為**（寬容模式）：
   - code 啟動時讀 `official_settings.contract_version`
   - 若 DB 版本 < code 期待版本 → `console.error` 警告 + **正常跑**（讀新欄位會 null → fallback defaults）
   - 若 DB 版本 > code 期待版本 → `console.error` + Telegram 告警（若有 TG bot 接）+ **正常跑**（忽略新欄位）
   - **不 fail-fast 擋 Bot**（避免升版過程全員離線）
5. **Rollback**：code 回退舊版（`git revert` + push），DB 欄位保留（下次升版再清，不 DROP）

### 11.5 禁止的升版操作

- ❌ 同一 migration 既 ADD 又 DROP / ALTER 型別
- ❌ 資料遷移不可 apply 到 production（先複製到 staging 驗過）
- ❌ code 讀新欄位不做 null check（會 crash）

## 附錄 C：Handoff Priority 決策表

略（同 v3 附錄 C + v4 完整 code 範例）。

### Webhook dispatch 實作（early-return，完整版）

```js
async function handleUserMessage(userId, text, replyToken) {
  const state = await getUserPathState(userId);

  // 封鎖用戶直接返回（v4 附錄 G）
  if (state.is_blocked) return;

  // Stage 5 特例：Bot 靜默 + notify
  if (state.path_stage === 5) return notifyWanxinOnStage5Reply(userId, text, state);

  // Level 0: 禮貌結束
  const politeKeys = await getSettingTyped('ai_polite_end_keywords');
  if (politeKeys.some(kw => text.includes(kw))) return handlePoliteEnd(userId, replyToken);

  // Level 1: 全域 Handoff（stage>=2）
  if (state.path_stage >= 2) {
    const reason = await matchGlobalHandoff(text);
    if (reason) return triggerHandoff(userId, reason, replyToken);
  }

  // Level 2/3: outro on_keyword_match（stage=4 等待）
  if (state.path_stage === 4) {
    const template = await getOutroTemplate(state.path);
    const match = matchOutroKeyword(template.on_keyword_match, text);
    if (match) return triggerHandoff(userId, match.reason, replyToken);
  }

  // Level 4: 正常路徑分類
  return routeByStage(state, text, replyToken);
}
```

## 附錄 G：邊緣情境宣告（v4 新增）

### G.1 封鎖 / 解封鎖

- `is_blocked=true` → webhook dispatch 第一步直接返回，不觸發任何流程
- 解封鎖後（`markBlocked(userId, false)`）→ 下次用戶主動回訊息走 stage=0 觸發 B（當作新用戶）
- **不保留封鎖前的 path_stage**（避免用戶回來被誤當 stage=3 繼續）

### G.2 官方帳號停權 / webhook URL 換

- 屬營運災備，不在契約 code 範圍
- 用戶累積的 ai_tags / 資料保留 90 天，換新帳號時舊資料作廢或手動遷移

### G.3 GDPR / 帳號刪除

- 婉馨後台 DELETE user 時，必須清：
  ```sql
  UPDATE official_line_users SET
    ai_tags = '{}'::jsonb,
    current_weight = NULL, target_weight = NULL,
    path = NULL, path_stage = 0,
    handoff_reason = NULL, handoff_triggered_at = NULL,
    ai_tags_updated_at = NULL,
    display_name = '[DELETED]'
  WHERE line_user_id = ?;
  ```
- **保留 `line_user_id` + `enrolled_from_path` + `enrolled_at`**（北極星量測需要）

### G.4 家人共用 LINE

- 一個 `line_user_id` 一個 record，Bot 無法區分同 userId 的多個使用者
- 已知限制，不處理
- 若用戶主動反映「這是我老婆的體重」→ 婉馨後台人工修正

### G.5 測驗做兩次

- `handleQuizCodeClaim` 呼叫時若 `path_stage != 0`：
  - **不推 q1_init**（不打斷對話）
  - 只更新 `metabolism_type`（新測驗結果覆蓋舊的）
  - 推代謝報告（新 metabolism_type 對應版本）
  - 不重置 `path_stage`
- 若 `path_stage = 0`：照正常流程觸發 A

### G.6 Path D 餐點含圖片

- 用戶傳 image 訊息 + 文字 < `meal_min_chars` 字 → 走 E6'：追問打字（AI 無 vision）
- 用戶傳純 image → 追問「能不能也打字說一下你常吃什麼？」
- 未來若接 vision model，契約升級時再加

### G.7 `is_active` 反向切換（true → false）

- 婉馨後台把某模板 is_active 從 true 改 false 後，正在 stage=4 等 outro 的用戶：
  - Webhook `getTemplate(path, stage, condition)` 會回 null（因為 partial index 只掃 is_active=true）
  - Fallback 邏輯：回「感謝你的回覆，我會請專人再跟你聊」+ 觸發 Handoff (reason=manual)
- 避免用戶看到 error
