1. **雙軌觸發**：被動（Q4 後用戶主動回訊息 + Q5_intent=continue）+ 主動（Q4 後 24h 無回應 → cron push）

4. **追蹤「點了 /apply 卻沒報名」是核心 KPI**
5. **真人保留處理猶豫**：handoff 訊息承諾具體時間 + 「不急推課」

export async function performQ5Transition({ userId, source, pushFn }) {
  const now = new Date().toISOString();
  const isActive = source === 'active';

  const updates = {
    path_stage: 6,
    path_stage_updated_at: now,
    q5_sent_at: now,

    .eq('line_user_id', userId)
    .is('q5_sent_at', null)
    .select('line_user_id, path_stage');

  if (updateErr) {
    console.error('[q5] UPDATE failed:', updateErr);
    return { ok: false, reason: 'db_error' };
  }
  if (!updated || updated.length === 0) {
    return { ok: false, reason: 'race_lost' };
  }

  const pushOk = await pushFn(userId);
  if (!pushOk) {
    // v2.3: rollback 加 path_stage=6 保護免 regress
    // rollback 本身若失敗（DB 不通）→ stage 卡 6 但沒推，靠 Ch.5.5 48h cron reset 兜底

      })
      .eq('line_user_id', userId)
      .eq('path_stage', 6);
    return { ok: false, reason: 'push_failed_rollback' };
  }

  return { ok: true };
}

### 0.9 🆕 /apply HMAC Signed URL 驗證（v2.4 升級）

**為什麼改 HMAC**：v2.3 計畫 LIFF userId 驗證，實作 PR #26/#30 實證做不到 — LINE 2024 policy 把 LIFF 綁在 Login channel，`liff.getContext().userId` 跟 Messaging API userId 是不同 namespace（刻意 hash 不同），server-side 比對永遠失敗。LIFF 簡化版（PR #30）只留 LIFF init 當 UX 骨架，authority 移到 HMAC signed URL。

**HMAC 擋什麼、不擋什麼**：

| 攻擊 | HMAC 擋得住？ |
|------|-------------|
| URL 造假（自己寫 `/apply?userid=U_stranger`）| ✅ sig 驗證失敗 reject |
| 過期 URL（24h 前 Bot 推的 URL）| ✅ ts 驗證失敗 reject |
| secret 洩漏 | 🔶 靠 key version rotate 換 secret（25h 雙驗期）|
| LINE-to-LINE 分享污染（朋友拿到 Q5 訊息截圖點連結填表）| ❌ 不擋，Phase 4.5 觀察 phone/email 重複率再評估 |

**0.9.1 URL 結構**

```
https://<apply_url_base>?userid=<U>&source=bot_q5&trigger=<passive|active>&kv=<key_version>&ts=<unix_sec>&sig=<base64url>
```

**0.9.2 Canonicalized Payload**（固定順序，避免 URL encode 差異）

```
kv=<kv>&source=<source>&ts=<ts>&trigger=<trigger>&userid=<userid>
```

- 按 key 字母序排列，不是 URL querystring raw
- 不含 `sig` 本身

**0.9.3 Secret 管理（Vercel env）**

```
Q5_APPLY_SIGNING_SECRET_V1=<random 64 bytes base64>
Q5_APPLY_SIGNING_SECRET_V2=<...>              ← rotate 時加
Q5_APPLY_SIGNING_KEY_VERSION=1                ← 當前版本（生成 URL 用）
Q5_APPLY_SIG_MAX_AGE_SEC=86400                ← URL 過期秒數，不走 DB 避免每次 verify 打 Supabase
```

**Rotate 流程（v2.4 修 C3 — 動態等待）**：

1. 加 `V2` env + **不改** `KEY_VERSION` → 雙驗期（V1 生成 + V1/V2 都能驗）
2. 改 `KEY_VERSION=2` → Vercel 自動 redeploy 觸發新 URL 用 V2 生，舊 URL（kv=1）仍可驗
3. **拔 V1 前**：先讀 `Q5_APPLY_SIG_MAX_AGE_SEC` 值（可能被改過），等待時間 = `max(max_age + 3600s, 25h)`
4. 等待時間結束後 → 移除 `V1` env → Vercel redeploy

**為什麼動態等待**：若管理者把 `Q5_APPLY_SIG_MAX_AGE_SEC` 調高（例如活動延長 URL 壽命到 7 天），固定 25h 拔 V1 會讓大量 in-flight URL 變 `invalid_signature`。動態讀避免踩坑。

**0.9.4 驗證邏輯**（`lib/q5-apply-url.js`，v2.4 修 C1 payload injection + M1 cache + C3 env rotate）

```js
import crypto from 'crypto';
import { getSettingTyped } from './official-settings.js';

// v2.4 白名單（修 Critical #1：payload injection 防禦）
const USERID_RE = /^U[0-9a-f]{32}$/;
const TS_RE = /^\d{10}$/;                    // 10 位 unix sec（涵蓋 2001~2286 年）
const SOURCE_ALLOWED = new Set(['bot_q5']);  // 未來擴 Set.add 就好
const TRIGGER_ALLOWED = new Set(['passive', 'active']);

export function buildQ5ApplyPayload({ userid, source, trigger, kv, ts }) {
  // 按 key 字母序固定排列（不用 URLSearchParams 避免 encode 差異）
  return `kv=${kv}&source=${source}&ts=${ts}&trigger=${trigger}&userid=${userid}`;
}

export async function buildQ5ApplyUrl({ userId, triggerSource }) {
  const base = await getSettingTyped('apply_url_base');
  const kv = parseInt(process.env.Q5_APPLY_SIGNING_KEY_VERSION || '1', 10);
  const secret = process.env[`Q5_APPLY_SIGNING_SECRET_V${kv}`];
  if (!secret) throw new Error(`q5_apply_signing_secret_v${kv}_missing`);
  const ts = Math.floor(Date.now() / 1000);
  const params = { userid: userId, source: 'bot_q5', trigger: triggerSource, kv, ts };
  const payload = buildQ5ApplyPayload(params);
  const sig = crypto.createHmac('sha256', secret).update(payload).digest('base64url');
  const qs = new URLSearchParams({ ...params, sig }).toString();
  return `${base}?${qs}`;
}

// v2.4: 同步函數（不 await DB），max_age 走 env 避免 verify 熱路徑打 DB
export function verifyQ5ApplySig({ userid, source, trigger, kv, ts, sig }) {
  // 1. 全欄位存在性
  if (!userid || !source || !trigger || !kv || !ts || !sig) {
    return { ok: false, reason: 'missing_param' };
  }
  // 2. v2.4 shape whitelist（修 C1 payload injection）
  if (typeof userid !== 'string' || !USERID_RE.test(userid)) {
    return { ok: false, reason: 'bad_userid_shape' };
  }
  if (typeof source !== 'string' || !SOURCE_ALLOWED.has(source)) {
    return { ok: false, reason: 'bad_source' };
  }
  if (typeof trigger !== 'string' || !TRIGGER_ALLOWED.has(trigger)) {
    return { ok: false, reason: 'bad_trigger' };
  }
  const kvStr = String(kv);
  if (!/^[1-9][0-9]?$/.test(kvStr)) {
    return { ok: false, reason: 'bad_key_version' };  // 1-99 只接 digit
  }
  const kvInt = parseInt(kvStr, 10);
  if (typeof ts !== 'string' && typeof ts !== 'number') {
    return { ok: false, reason: 'bad_ts_type' };
  }
  const tsStr = String(ts);
  if (!TS_RE.test(tsStr)) return { ok: false, reason: 'bad_ts_shape' };
  if (typeof sig !== 'string' || sig.length < 20 || sig.length > 100) {
    return { ok: false, reason: 'bad_sig_shape' };
  }

  // 3. key version 可識別
  const secret = process.env[`Q5_APPLY_SIGNING_SECRET_V${kvInt}`];
  if (!secret) return { ok: false, reason: 'unknown_key_version' };

  // 4. HMAC 驗證（timingSafeEqual 前先 length 保護）
  const payload = buildQ5ApplyPayload({ userid, source, trigger, kv: kvInt, ts: tsStr });
  const expected = crypto.createHmac('sha256', secret).update(payload).digest('base64url');
  const expectedBuf = Buffer.from(expected);
  const sigBuf = Buffer.from(sig);
  if (expectedBuf.length !== sigBuf.length) return { ok: false, reason: 'bad_sig' };
  if (!crypto.timingSafeEqual(expectedBuf, sigBuf)) return { ok: false, reason: 'bad_sig' };

  // 5. 時間檢查（走 env，不打 DB）
  const tsInt = parseInt(tsStr, 10);
  const now = Math.floor(Date.now() / 1000);
  const maxAge = parseInt(process.env.Q5_APPLY_SIG_MAX_AGE_SEC || '86400', 10);
  if (now - tsInt > maxAge) return { ok: false, reason: 'expired' };
  if (tsInt - now > 300) return { ok: false, reason: 'future_ts' }; // clock skew 5min
  return { ok: true };
}
```

**0.9.5 驗證時機 + Route Handler 實作（v2.4 修 C2 + M2）**

| Endpoint | 驗 HMAC？ | 理由 |

**Client 呼叫**：`/apply` 頁 LIFF init 成功後，用 `fetch('/api/apply/visit', { method: 'POST', headers: {'Content-Type':'application/json'}, body: JSON.stringify({userid, source, trigger, kv, ts, sig}) })` 送 6 個欄位（從 URL params 原封不動帶過去，不做前端轉型）。

**Route handler 共用 pattern**（v2.4 修 C2 — 防 query/body 重複 key 繞過）：

```js
// /app/api/apply/visit/route.js
import { verifyQ5ApplySig } from '@/lib/q5-apply-url';

export async function POST(req) {
  let body;
  try {
    body = await req.json();
  } catch {
    return Response.json({ error: 'invalid_body' }, { status: 400 });
  }

  // 所有欄位必須是 primitive（擋 array injection: body={userid:['A','B']}）
  const keys = ['userid', 'source', 'trigger', 'kv', 'ts', 'sig'];
  for (const k of keys) {
    const v = body[k];
    if (v === undefined || v === null) {
      return Response.json({ error: 'invalid_signature' }, { status: 400 });
    }
    if (typeof v !== 'string' && typeof v !== 'number') {
      return Response.json({ error: 'invalid_signature' }, { status: 400 });
    }
  }

  const verifyResult = verifyQ5ApplySig(body);
  if (!verifyResult.ok) {
    // 內部 log 真實 reason，對外統一 invalid_signature（避免 enum）
    console.warn('[apply/visit] verify failed:', verifyResult.reason, { userid: body.userid });
    return Response.json({ error: 'invalid_signature' }, { status: 400 });
  }

  // verified，繼續業務邏輯（stage 6→7 + click_count）
  // ...
}
```

**為什麼走 body 不走 query**：
1. body primitive check 簡單擋 array（query `?userid=A&userid=B` 依 parser 行為而異）
2. URL query 可能被 proxy / log / referrer 洩漏，body 相對乾淨
3. POST body 跟 GET query 在 Next.js 14 的 parse 行為不同，集中在 body 降低心智負擔

**URL 產生到驗證的完整流**：

```
Bot pushQ5SoftInvite → buildQ5ApplyUrl 產 URL → LINE Quick Reply 按鈕 URI
用戶點 → 瀏覽器進 /apply?userid=X&... → LIFF init → fetch POST /api/apply/visit 帶 6 欄位 body
Route handler 做 body shape check → verifyQ5ApplySig → 通過才 UPDATE stage/click_count
```

**0.9.6 URL 生成點**

`lib/q5-message.js: pushQ5SoftInvite(userId, triggerSource)` 內部呼叫 `buildQ5ApplyUrl({ userId, triggerSource })`。Phase 4.2 wire 時才實際推送，Phase 4.1 先備好 helper + unit test。

**0.9.7 Error Response 政策**

所有 verify 失敗回統一 `{ error: 'invalid_signature' }` + HTTP 400，不暴露具體 reason（避免攻擊者 enum secret version / payload 結構）。內部 log 完整 reason 方便 debug。

**0.9.8 backward compat**

Phase 4.1 上線前無活的 Q5 URL（Phase 4.2 才 wire pushQ5SoftInvite），不存在舊 URL 衝突。Phase 4.2 Q5 正式推送才開始產生 signed URL。

### 1.1 狀態定義

| path_stage | 意義 | path 值 |
|-----------|------|---------|
| 0 | 未進入 / 已 reset | NULL |
| 1-4 | Q1-Q4 | 既有 |
| 5 | Handoff 已觸發 | 保留 |
| 6 | Q5 軟邀請已推送 | 保留 |
| 7 | /apply 已訪問 | 保留 |
| 8 | 已報名 | 保留 |

### 1.2 狀態轉換圖

```
                Q4 AI 綜合回饋已送達
                path_stage=4
                     │
         ┌───────────┴────────────┐
         │                        │
   【被動軌】                  【主動軌】
   用戶下一則訊息              Q4 送達後 24h
   非 handoff/polite end       用戶零回應
         │                        │
         ▼                        ▼
   Q5_intent_classifier     cron/q5-maintenance
   AI→ continue/decline/     主動軌處理（無 AI）
      ai_failed
         │                        │
    continue                      │
         └────────┬───────────────┘
                  ▼
         performQ5Transition（Ch.0.8 helper）
         - atomic UPDATE WHERE q5_sent_at IS NULL
         - UPDATE OK → pushMessage
         - push 失敗 → rollback（含 path_stage=6 保護）
                  │
                  ▼
         stage=6
         （被動軌另 updateQ5Intent('continue')）
                  │
         ┌────────┴────────┐
         ▼                 ▼
   按「看看做法」       按「有問題想問」
   (URI → /apply + LIFF 驗證)  (postback)
         │                 │
         ▼                 ▼
   POST /api/apply/visit  triggerHandoff
   stage=7                reason='q5_followup'
   q5_clicked_at          stage=5 + 專屬文案
         │
         ▼
   用戶在 /apply 填表
         │
     ┌───┴───┐
     ▼       ▼
   submit   離開
     │       │
     ▼       ▼
   RPC（含  cron/q5-visit-followup
   IF NOT  （24h 後推 1 次）
    FOUND）
   stage=8
```

### 1.3 關鍵狀態邊界

- handoff 覆蓋 Q5：stage=6 → stage=5
- 雙軌 atomic：`WHERE q5_sent_at IS NULL`
- 主動軌 skip：`q5_intent IN ('decline', 'ai_failed')`
- 被動軌 AI fallback：5 種 error path 都寫 `ai_failed`
- stage=6 無反應 48h → cron reset
- stage=7 → cron 不 reset，改 visit-followup
- stage=8：終態

### 3.1 Q5_intent_classifier（v2.3 補 5 個 error path）

**Prompt**（維持 v2.0，continue / decline）

**Fallback 完整列表**：

| Error path | 處理 |
|-----------|------|
| `gemini_timeout` | updateQ5Intent('ai_failed') |
| `gemini_api_{status}`（429 / 401 etc.）| updateQ5Intent('ai_failed') |
| `gemini_no_text`（safety filter）| updateQ5Intent('ai_failed') |
| `gemini_no_key` | updateQ5Intent('ai_failed') + telegram 告警（環境問題）|
| `gemini_json_parse` | updateQ5Intent('ai_failed') |
| validator reject | updateQ5Intent('ai_failed') |
| updateQ5Intent 自身失敗 | log，主動軌 SQL race guard 擋住二次推送 |

**實作 pattern**：抄 `lib/ai-classifier.js:583-604 verifyHandoffIntent` 的 try-catch。

**Validator**：

```js
function validateQ5Intent(output) {
  if (!output || typeof output !== 'object') return { valid: false };
  if (!['continue', 'decline'].includes(output.intent)) return { valid: false };
  if (!['high', 'medium', 'low'].includes(output.confidence)) return { valid: false };
  if (typeof output.reason !== 'string') return { valid: false };
  return { valid: true, data: output };
}
```

### 3.2 Q5 軟邀請訊息（v2.3 B1c 保守版）

**被動軌**（`q5_soft_invite_passive_text`）：

## 5. 觸發規則表

### 5.1a 被動軌

```
觸發：stage=4 用戶傳文字
條件：
  1. 不命中 polite end / handoff
  2. 文字長度 ≥ 2 字
  3. q5_intent IS NULL
動作：
  1. Q5_intent_classifier → intent
  2. intent='continue':
     → await performQ5Transition({ userId, source: 'passive', pushFn })
     → await updateQ5Intent(userId, 'continue')
  3. intent='decline':
     → await updateQ5Intent(userId, 'decline')
     → 靜默或 polite end
  4. 5 種 AI error:
     → await updateQ5Intent(userId, 'ai_failed')
     → 保持 stage=4
```

### 5.1b 主動軌

```
觸發：cron/q5-maintenance 主動軌段
條件（SQL WHERE）：
  stage = 4
  AND (ai_tags->>'q4_classified_at')::timestamptz < now() - INTERVAL '24 hours'
  AND q5_sent_at IS NULL
  AND (q5_intent IS NULL OR q5_intent NOT IN ('decline', 'ai_failed'))
  AND last_user_reply_at < (ai_tags->>'q4_classified_at')::timestamptz
  AND (ai_tags->>'意願' IS NULL OR ai_tags->>'意願' != 'low')
  AND line_user_id NOT IN (TEST_ALLOWLIST)（q5_test_mode_cron = true）
動作（每用戶）：
  1. performQ5Transition({ userId, source: 'active', pushFn })
  2. if (!result.ok) log + skip
  3. 成功則 helper 已寫 stage=6 + q5_sent_at + q5_active_invite_sent_at
不重試：helper rollback 處理
```

### 5.2 Stage=6 → Stage=7（點 /apply，v2.4 改 HMAC）

```
觸發：點 Quick Reply「看看做法」→ /apply（LIFF init only）→ POST /api/apply/visit
條件：
  1. verifyQ5ApplySig 通過（Ch.0.9.4）
  2. userId 存在於 official_line_users
  3. stage >= 6
動作：
  1. HMAC 驗證失敗 → HTTP 400 { error: 'invalid_signature' }，不改 stage
  2. 驗證通過 → UPDATE stage=7（若 stage=6），q5_clicked_at=COALESCE, q5_click_count += 1
  3. 回 OK
```

**Client 呼叫**：/apply 頁 LIFF init 成功後，由 React useEffect 以 POST JSON body 送 `{userid, source, trigger, kv, ts, sig}`（從 URL params 原封不動帶過去）。

### 5.3 Stage=7 → Stage=8（v2.3 加 IF NOT FOUND RAISE，v2.4 加 HMAC 前置）

**前置**：`/api/apply/submit` 先跑 `verifyQ5ApplySig`，失敗直接 HTTP 400 `{ error: 'invalid_signature' }`，不進 RPC。驗證通過才執行下方 PL/pgSQL。

**PL/pgSQL function**：

```sql
DROP FUNCTION IF EXISTS submit_application CASCADE;
DROP TYPE IF EXISTS submit_application_result;

CREATE TYPE submit_application_result AS (
  application_id BIGINT,
  enrolled_at TIMESTAMPTZ,
  other_apps_count INTEGER,
  other_phone_count INTEGER
);

CREATE OR REPLACE FUNCTION submit_application(
  p_line_user_id TEXT,
  p_real_name TEXT,
  p_phone TEXT,
  p_email TEXT,
  p_address TEXT,
  p_gender TEXT,
  p_age INTEGER,
  p_line_id TEXT,
  p_display_name TEXT,
  p_program_choice TEXT,
  p_agreed_refund_policy BOOLEAN,
  p_source TEXT
) RETURNS submit_application_result
LANGUAGE plpgsql
AS $$
DECLARE
  v_app_id BIGINT;
  v_user_path TEXT;
  v_enrolled_at TIMESTAMPTZ;
  v_other_apps INTEGER;
  v_other_phone INTEGER;
BEGIN
  -- v2.3: 先驗用戶存在
  SELECT path INTO v_user_path FROM official_line_users WHERE line_user_id = p_line_user_id;
  IF NOT FOUND THEN
    RAISE EXCEPTION 'user_not_found: %', p_line_user_id USING ERRCODE = 'P0002';
  END IF;

  INSERT INTO official_program_applications (
    line_user_id, real_name, phone, email, address, gender, age,
    line_id, display_name, program_choice, agreed_refund_policy,
    source, status, notify_status
  ) VALUES (
    p_line_user_id, p_real_name, p_phone, p_email, p_address, p_gender, p_age,
    p_line_id, p_display_name, p_program_choice, p_agreed_refund_policy,
    p_source, 'pending', 'pending'
  ) RETURNING id INTO v_app_id;

  UPDATE official_line_users
  SET
    path_stage = 8,
    enrolled_at = COALESCE(enrolled_at, now()),
    enrolled_from_path = COALESCE(enrolled_from_path, v_user_path)
  WHERE line_user_id = p_line_user_id
  RETURNING enrolled_at INTO v_enrolled_at;

  SELECT COUNT(*) INTO v_other_apps
  FROM official_program_applications
  WHERE line_user_id = p_line_user_id AND id != v_app_id;

  SELECT COUNT(*) INTO v_other_phone
  FROM official_program_applications
  WHERE phone = p_phone AND id != v_app_id;

  RETURN (v_app_id, v_enrolled_at, v_other_apps, v_other_phone)::submit_application_result;
END;
$$;
```

## 12. 量測定義字典

### 12.1 核心指標

與 v2.2 同（9 個指標）。

### 12.1a click_rate 量測邊界（v2.4 新增）

HMAC signed URL 只證明「這 URL 是 Bot 生的」，不證明「點擊者是 userid 本人」。以下情境會污染 `q5_click_count`，**但不污染 `q5_clicked_at` 當「首次點擊」的 unique 量測**：

| 污染源 | 影響 `q5_clicked_at`（unique）？ | 影響 `q5_click_count`（總計）？ |
|-------|--------------------------------|--------------------------------|
| LINE-to-LINE 分享（朋友點）| ⚠️ 會，會讓「本人沒看到」的用戶錯標 clicked | ✅ 會累加 |
| LINE 預覽 bot / LINE iOS 連結預覽 | 不會（用 COALESCE 固定首次）| 可能累加 |
| 一休 admin 自己點 | 可能（若 admin 不走白名單）| 會累加 |
| 瀏覽器手動訪問 URL | 會 | 會累加 |

**量測選擇**：Phase 4.5 北極星 `click_rate by path` 用 `q5_clicked_at IS NOT NULL` 當分子（unique 點擊），接受 LINE-to-LINE 分享造成的少量誤差（一休 2026-04-23 決策）。`q5_click_count` 僅供輔助觀察「分享熱度」間接指標。

**若 Phase 4.5 觀察期發現分享污染嚴重**（phone/email 重複率 > 5%）→ 升級選項：

1. client 端加 `liff.isInClient()` 判斷，只有 true 才 POST visit（但桌面 LINE 會回 false，false negative 風險）
2. 加手機驗證碼到 submit（硬根治 LINE-to-LINE 分享）
3. 加 `q5_unique_submitter_phone` 等欄位記錄真正報名者 phone hash 做後期 de-dup

### 12.2 觀察期前提 + Gate

**前提條件**：
1. TEST_MODE=false ≥ 1 週
2. Q1-Q4 漏斗 baseline（每天到 stage=4 人數）

4. **LIFF 驗證生效**（Ch.0.9，確保不被盜用污染 baseline）

**Gate 原則**：不拍腦袋，先量 baseline + 95% 置信區間。

**TEST_MODE=false 開關**（v2.3 補 Pattern 5）：
